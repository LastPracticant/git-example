```bash
# Клонируем репозиторий
git clone https://github.com/LastPracticant/ya-practicum-m2.git

# Переходим в папку
cd ya-practicum-m2

# Переключаемся на dev-ветку
## Флаг `-b` означает "Создать новую ветку". Если ветка у вас уже существует, `-b` писать не нужно
git checkout -b dev

# Спуливаем последние изменения
## git pull --rebase [remote-name] [branch-name]
git pull --rebase origin dev

# Переключаемся на новую ветку
git checkout -b LP-19

# Пишем код

# Индексируем изменения (`.` означает все файлы)
git add .

# Фиксируем (коммитим) изменения
## кто-то в коммитах пишет, отвечая на вопрос "Что я сделал?" — "Добавил главную страницу"
## кто-то пишет, отвечая на вопрос "Что делает этот коммит?" — "Добавляет главную страницу"
## не принципиально, как именно писать, зафиксируйте себе в голове один из вопросов — меньше времени будет тратиться на обдумывание
git commit -m 'LP-19: добавляет главную страницу'

# На всякий случай пуллимся и ребейзимся на dev еще раз
git pull --rebase origin dev

# Пушим коммиты на удаленный репозиторий
## git push [remote-name] [branch-name]
git push origin LP-19

# Делаем Pull Request на Github'e, призываем ревьюеров, ждем ревью

# Вносим изменения в код после код-ревью

# Снова индексируем, коммитим и пушим
## Ребейзиться на dev после код-ревью нежелательно, поскольку после ребейза коммиты будут новые и нужно будет пушить с форсом, а это нарушит логику для ревьюеров в ПР
git add .
git commit -m 'LP-19: правит сигнатуру компонента маршрутизации'
git add .
git commit -m 'LP-19: добавляет generic-интерфейс для описания initialData'
git push origin LP-19

# После прохождения код-ревью хорошо бы поребейзиться и запушить заново, если прилетели новые коммиты из dev
git pull --rebase origin dev
git push --force origin LP-19 # Если новых коммитов не было, пушить не нужно

# В Pull Request'е на гитхабе нажимаем кнопку Squash and merge, редактируем описание и мержим

# Нажимаем рядом кнопку Delete branch. Мы удалили ветку в remote

# Теперь хорошо бы переключиться на dev, удалить вашу ветку локально и спуллить новый dev
git checkout dev
git pull --rebase origin dev
git branch -D LP-19
```
